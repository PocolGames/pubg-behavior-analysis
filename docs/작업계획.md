```css
  --pubg-green: #7ed321;        /* 성공, 생존 */
```

   ```css
   .pubg-input {

     box-shadow: inset 0 0 5px rgba(74, 158, 255, 0.2);
   }
   ```

   ```css
   .btn-pubg {

     text-transform: uppercase;
     font-weight: bold;
     letter-spacing: 1px;
   ```
